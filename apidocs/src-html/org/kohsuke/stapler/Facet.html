024package org.kohsuke.stapler;
025
026import org.apache.commons.discovery.ResourceNameIterator;
027import org.apache.commons.discovery.resource.ClassLoaders;
028import org.apache.commons.discovery.resource.names.DiscoverServiceNames;
029import org.kohsuke.MetaInfServices;
030import org.kohsuke.stapler.lang.Klass;
031
032import javax.servlet.RequestDispatcher;
033import javax.servlet.ServletException;
034import java.io.IOException;
035import java.util.ArrayList;
036import java.util.HashSet;
037import java.util.List;
038import java.util.Set;
039import java.util.logging.Level;
040import java.util.logging.Logger;

042/**
043 * Aspect of stapler that brings in an optional language binding.
044 *
045 * Put {@link MetaInfServices} on subtypes so that Stapler can discover them.
046 *
047 * @author Kohsuke Kawaguchi
048 */
049public abstract class Facet {
050    /**
051     * Adds {@link Dispatcher}s that look at one token and binds that
052     * to the views associated with the 'it' object.
053     */
054    public abstract void buildViewDispatchers(MetaClass owner, List<Dispatcher> dispatchers);
055
056    /**
057     * Adds {@link Dispatcher}s that serves the likes of {@code index.EXT}
058     *
059     * The default implementation invokes {@link #handleIndexRequest(RequestImpl, ResponseImpl, Object, MetaClass)}
060     * but facet implementations can improve runtime dispatch performance by testing the presence
061     * of index view page upfront.
062     */
063    public void buildIndexDispatchers(MetaClass owner, List<Dispatcher> dispatchers) {
064        dispatchers.add(new IndexViewDispatcher(owner,this));
065    }
066
067    /**
068     * Adds {@link Dispatcher}s that do catch-all behaviours like "doDispatch" does.
069     */
070    public void buildFallbackDispatchers(MetaClass owner, List<Dispatcher> dispatchers) {}
071
072    /**
073     * Discovers all the facets in the classloader.
074     */
075    public static List<Facet> discover(ClassLoader cl) {
076        return discoverExtensions(Facet.class, cl);
077    }
078
079    public static <T> List<T> discoverExtensions(Class<T> type, ClassLoader... cls) {
080        List<T> r = new ArrayList<T>();
081        Set<String> classNames = new HashSet<String>();
082
083        for (ClassLoader cl : cls) {
084            ClassLoaders classLoaders = new ClassLoaders();
085            classLoaders.put(cl);
086            DiscoverServiceNames dc = new DiscoverServiceNames(classLoaders);
087            ResourceNameIterator itr = dc.findResourceNames(type.getName());
088            while(itr.hasNext()) {
089                String name = itr.nextResourceName();
090                if (!classNames.add(name))  continue;   // avoid duplication
091                
092                Class<?> c;
093                try {
094                    c = cl.loadClass(name);
095                } catch (ClassNotFoundException e) {
096                    LOGGER.log(Level.WARNING, "Failed to load "+name,e);
097                    continue;
098                }
099                try {
100                    r.add((T)c.newInstance());
101                } catch (InstantiationException e) {
102                    LOGGER.log(Level.WARNING, "Failed to instantiate "+c,e);
103                } catch (IllegalAccessException e) {
104                    LOGGER.log(Level.WARNING, "Failed to instantiate "+c,e);
105                }
106            }
107        }
108        return r;
109    }
110
111    public static final Logger LOGGER = Logger.getLogger(Facet.class.getName());
112
113    public static boolean ALLOW_VIEW_NAME_PATH_TRAVERSAL = Boolean.getBoolean(Facet.class.getName() + ".allowViewNamePathTraversal");
114    
115    /**
116     * Creates a {@link RequestDispatcher} that handles the given view, or
117     * return null if no such view was found.
118     *
119     * @param type
120     *      If "it" is non-null, {@code it.getClass()}. Otherwise the class
121     *      from which the view is searched.
122     */
123    public RequestDispatcher createRequestDispatcher(RequestImpl request, Klass<?> type, Object it, String viewName) throws IOException {
124        return null;    // should be really abstract, but not
125    }
126
127    public RequestDispatcher createRequestDispatcher(RequestImpl request, Class type, Object it, String viewName) throws IOException {
128        return createRequestDispatcher(request,Klass.java(type),it,viewName);
129    }
130
131    /**
132     * Attempts to route the HTTP request to the 'index' page of the 'it' object.
133     *
134     * @return
135     *      true if the processing succeeds. Otherwise false.
136     */
137    public abstract boolean handleIndexRequest(RequestImpl req, ResponseImpl rsp, Object node, MetaClass nodeMetaClass) throws IOException, ServletException;
138
139    /**
140     * Maps an instance to a {@link Klass}. This is the generalization of {@code o.getClass()}.
141     * 
142     * This is for those languages that use something other than {@link Class} to represent the concept of a class.
143     * Those facets that are fine with {@code o.getClass()} should return null so that it gives other facets a chance
144     * to map it better.
145     */
146    public Klass<?> getKlass(Object o) {
147        return null;
148    }
149    
150    /**
151     * Ensure the path that is passed is only the name of the file and not a path
152     */
153    protected boolean isBasename(String potentialPath){
154        if (ALLOW_VIEW_NAME_PATH_TRAVERSAL) {
155            return true;
156        } else {
157            if (potentialPath.contains("\\") || potentialPath.contains("/")) {
158                // prevent absolute path and folder traversal to find scripts
159                return false;
160            }
161            
162            return true;
163        }
164    }
