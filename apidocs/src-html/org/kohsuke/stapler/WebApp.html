043/**
044 * Object scoped to the entire webapp. Mostly used for configuring behavior of Stapler.
045 *
046 * <p>
047 * In contrast, {@link Stapler} is a servlet, so there can be multiple instances per webapp.
048 *
049 * @author Kohsuke Kawaguchi
050 * @see WebApp#get(ServletContext)
051 * @see WebApp#getCurrent()  
052 * @see Stapler#getWebApp()
053 */
054public class WebApp {
055    /**
056     * Obtains the {@link WebApp} associated with the given {@link ServletContext}.
057     */
058    public static WebApp get(ServletContext context) {
059        Object o = context.getAttribute(WebApp.class.getName());
060        if(o==null) {
061            synchronized (WebApp.class) {
062                o = context.getAttribute(WebApp.class.getName());
063                if(o==null) {
064                    o = new WebApp(context);
065                    context.setAttribute(WebApp.class.getName(),o);
066                }
067            }
068        }
069        return (WebApp)o;
070    }
071

073     * {@link ServletContext} for this webapp.

130     * Provides access to {@link Stapler} servlet instances. This is useful

144    public WebApp(ServletContext context) {
145        this.context = context;
146        // TODO: allow classloader to be given?
147        facets.addAll(Facet.discoverExtensions(Facet.class, Thread.currentThread().getContextClassLoader(), getClass().getClassLoader()));
148        responseRenderers.add(new HttpResponseRenderer.Default());
149    }

154     */

157    }

175    public ClassLoader getClassLoader() {
176        ClassLoader cl = classLoader;
177        if(cl==null)
178            cl = Thread.currentThread().getContextClassLoader();
179        if(cl==null)
180            cl = Stapler.class.getClassLoader();
181        return cl;
182    }

184    /**
185     * If the facet of the given type exists, return it. Otherwise null.
186     */
187    public <T extends Facet> T getFacet(Class<T> type) {
188        for (Facet f : facets)
189            if(type==f.getClass())
190                return type.cast(f);
191        return null;
192    }

194    /**
195     * Sets the classloader used by {@link StaplerRequest#bindJSON(Class, JSONObject)} and its sibling methods.
196     */
197    public void setClassLoader(ClassLoader classLoader) {
198        this.classLoader = classLoader;
199    }
200
201    public MetaClass getMetaClass(Class c) {
202        return getMetaClass(Klass.java(c));
203    }
204    
205    public MetaClass getMetaClass(Klass<?> c) {
206        if(c==null)     return null;
207        synchronized(classMap) {
208            MetaClass mc = classMap.get(c);
209            if(mc==null) {
210                mc = new MetaClass(this,c);
211                classMap.put(c,mc);
212            }
213            return mc;
214        }
215    }
216
217    /**
218     * Obtains a {@link MetaClass} that represents the type of the given object.
219     *
220     * <p>
221     * This code consults all facets to handle scripting language objects correctly.
222     */
223    public MetaClass getMetaClass(Object o) {
224        return getMetaClass(getKlass(o));
225    }
226
227    public Klass<?> getKlass(Object o) {
228        if (o instanceof KInstance) {
229            KInstance ki = (KInstance) o;
230            Klass k = ki.getKlass();
231            if (k!=null)
232                return k;
233        }
234
235        for (Facet f : facets) {
236            Klass<?> k = f.getKlass(o);
237            if (k!=null)
238                return k;
239        }
240        return Klass.java(o.getClass());
241    }
242    
243    /**
244     * Convenience maintenance method to clear all the cached scripts for the given tearoff type.
245     *
246     * <p>
247     * This is useful when you want to have the scripts reloaded into the live system without
248     * the performance penalty of {@link MetaClass#NO_CACHE}.
249     *
250     * @see MetaClass#NO_CACHE
251     */
252    public void clearScripts(Class<? extends AbstractTearOff> clazz) {
253        synchronized (classMap) {
254            for (MetaClass v : classMap.values()) {
255                AbstractTearOff t = v.getTearOff(clazz);
256                if (t!=null)
257                    t.clearScripts();
258            }
259        }
260    }
261
262    void addStaplerServlet(String servletName, Stapler servlet) {
263        if (servletName==null)  servletName=""; // be defensive
264        servlets.put(servletName,servlet);
265    }
266
267    /**
268     * Gets a reference to some {@link Stapler} servlet in this webapp.
269     *
270     * <p>
271     * Most Stapler webapps will have one {@code <servlet>} entry in web.xml
272     * and if that's the case, that'd be returned. In a fully general case,
273     * a webapp can have multiple servlets and more than one of them can be
274     * {@link Stapler}. This method returns one of those. Which one gets
275     * returned is unspecified.
276     *
277     * <p>
278     * This method is useful if you are in a {@link Filter} and using
279     * Stapler to handle the current request. For example,
280     *
281     * <pre>
282     * WebApp.get(servletContext).getSomeStapler().invoke(
283     *     request,response,
284     *     someJavaObject,
285     *     "/path/to/dispatch/request");
286     * </pre>
287     */
288    public Stapler getSomeStapler() {
289        return servlets.values().iterator().next();
290    }

292    /**
293     * Gets the current {@link WebApp} that the calling thread is associated with.
294     */
295    public static WebApp getCurrent() {
296        return Stapler.getCurrent().getWebApp();
297    }

299    public JsonInErrorMessageSanitizer getJsonInErrorMessageSanitizer() {
300        if (jsonInErrorMessageSanitizer == null) {
301            return JsonInErrorMessageSanitizer.NOOP;
302        }
303        return jsonInErrorMessageSanitizer;
304    }

306    /**
307     * Allow the application to customize the way the JSON are rendered in the stack trace in case of binding exception.
308     */
309    public void setJsonInErrorMessageSanitizer(JsonInErrorMessageSanitizer jsonInErrorMessageSanitizer) {
310        this.jsonInErrorMessageSanitizer = jsonInErrorMessageSanitizer;
311    }
