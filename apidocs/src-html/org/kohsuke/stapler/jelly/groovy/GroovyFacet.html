049/**
050 * {@link Facet} that brings in Groovy support on top of Jelly.
051 * 
052 * @author Kohsuke Kawaguchi
053 */
054@MetaInfServices(Facet.class)
055public class GroovyFacet extends Facet implements JellyCompatibleFacet {
056    public void buildViewDispatchers(final MetaClass owner, List<Dispatcher> dispatchers) {
057        dispatchers.add(new Dispatcher() {
058            final GroovyClassTearOff tearOff = owner.loadTearOff(GroovyClassTearOff.class);
059            final GroovyServerPageTearOff gsp = owner.loadTearOff(GroovyServerPageTearOff.class);
060
061            public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
062                // check Groovy view
063                String next = req.tokens.peek();
064                if(next==null)  return false;
065
066                // only match the end of the URL
067                if (req.tokens.countRemainingTokens()>1)    return false;
068                // and avoid serving both "foo" and "foo/" as relative URL semantics are drastically different
069                if (req.getRequestURI().endsWith("/"))      return false;
070
071                if (!isBasename(next)) {
072                    // potentially an attempt to make a folder traversal
073                    return false;
074                }
075
076                try {
077                    Script script = tearOff.findScript(next);
078                    if(script==null)
079                        script = gsp.findScript(next);
080                    if (script==null)
081                        return false;   // no Groovy script found
082
083                    req.tokens.next();
084
085                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s: Groovy facet: %s", next + ".groovy");
086                    if(traceable())
087                        trace(req,rsp,"Invoking "+next+".groovy"+" on "+node+" for "+req.tokens);
088
089                    WebApp.getCurrent().getFacet(JellyFacet.class).scriptInvoker.invokeScript(req, rsp, script, node);
090
091                    return true;
092                } catch (RuntimeException e) {
093                    throw e;
094                } catch (IOException e) {
095                    throw e;
096                } catch (Exception e) {
097                    throw new ServletException(e);
098                }
099            }
100            public String toString() {
101                // or TOKEN.gsp
102                return "TOKEN.groovy for url=/TOKEN";
103            }
104        });
105    }
106
107    public Collection<Class<GroovyClassTearOff>> getClassTearOffTypes() {
108        return TEAROFF_TYPES;
109    }
110
111    public Collection<String> getScriptExtensions() {
112        return EXTENSION;
113    }
114
115    public RequestDispatcher createRequestDispatcher(RequestImpl request, Klass type, Object it, String viewName) throws IOException {
116        MetaClass mc = request.stapler.getWebApp().getMetaClass(type);
117        return createDispatcher(it, viewName, mc);
118    }
119
120    private RequestDispatcher createDispatcher(Object it, String viewName, MetaClass mc) throws IOException {
121        RequestDispatcher d = mc.loadTearOff(GroovyClassTearOff.class).createDispatcher(it, viewName);
122        if (d==null)
123            d = mc.loadTearOff(GroovyServerPageTearOff.class).createDispatcher(it, viewName);
124        return d;
125    }
126
127    @Override
128    public void buildIndexDispatchers(MetaClass owner, List<Dispatcher> dispatchers) {
129        try {
130            if (owner.loadTearOff(JellyClassTearOff.class).findScript("index")!=null) {
131                super.buildIndexDispatchers(owner, dispatchers);
132            }
133        } catch (JellyException e) {
134            LOGGER.log(Level.WARNING, "Failed to parse index.groovy for "+owner, e);
135        }
136    }
137
138    public boolean handleIndexRequest(RequestImpl req, ResponseImpl rsp, Object node, MetaClass nodeMetaClass) throws IOException, ServletException {
139        RequestDispatcher d = createDispatcher(node,"index", nodeMetaClass);
140
141        if (d!=null) {
142            d.forward(req, rsp);
143            return true;
144        }
145        return false;
146    }
147
148    private static final Set<Class<GroovyClassTearOff>> TEAROFF_TYPES = Collections.singleton(GroovyClassTearOff.class);
149
150    private static final Set<String> EXTENSION = Collections.singleton(".groovy");
151}
