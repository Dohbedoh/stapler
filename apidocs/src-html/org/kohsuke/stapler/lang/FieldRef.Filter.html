001package org.kohsuke.stapler.lang;
002
003import org.apache.commons.lang.StringUtils;
004
005import java.lang.annotation.Annotation;
006import java.lang.reflect.Field;
007import java.lang.reflect.Modifier;
008import java.util.Arrays;
009import java.util.stream.Collectors;
010
011/**
012 * Fields of {@link Klass}.
013 *
014 * @see Klass#getDeclaredFields()
015 * @author Kohsuke Kawaguchi
016 */
017public abstract class FieldRef extends AnnotatedRef {
018
019    public interface Filter {
020        boolean keep(FieldRef m);
021
022        Filter ALWAYS_OK = new Filter() {
023            @Override
024            public boolean keep(FieldRef m) {
025                return true;
026            }
027        };
028    }
029
030    /**
031     * Name of the method.
032     *
033     * @see Field#getName()
034     */
035    public abstract String getName();
036
037    /**
038     * Whether the field is static.
039     * @return
040     */
041    public abstract boolean isStatic();
042
043    /**
044     * Obtains the value of the field of the instance.
045     */
046    public abstract Object get(Object instance) throws IllegalAccessException;
047
048    /**
049     * Gets a fully qualified name of this field that includes the declaring type.
050     */
051    public abstract String getQualifiedName();
052
053    /**
054     * Gets the signature for this for use in lists
055     *
056     * @see org.kohsuke.stapler.Function#getSignature()
057     *
058     */
059    public abstract String getSignature();
060
061    // TODO Should this be Klass?
062    public abstract Class<?> getReturnType();
063
064    /**
065     * Returns true if this method is a 'public' method that should be used for routing requests.
066     */
067    public boolean isRoutable() {
068        return true;
069    }
070
071    public static FieldRef wrap(final Field f) {
072        f.setAccessible(true);
073
074        return new FieldRef() {
075            @Override
076            public <T extends Annotation> T getAnnotation(Class<T> type) {
077                return f.getAnnotation(type);
078            }
079
080            @Override
081            public String getName() {
082                return f.getName();
083            }
084
085            @Override
086            public Object get(Object instance) throws IllegalAccessException {
087                return f.get(instance);
088            }
089
090            @Override
091            public boolean isStatic() {
092                return Modifier.isStatic(f.getModifiers());
093            }
094
095            @Override
096            public Class<?> getReturnType() {
097                return f.getType();
098            }
099
100            @Override
101            public String getSignature() {
102                String prefix = isStatic() ? "staticField" : "field";
103                return StringUtils.join(Arrays.asList(prefix, f.getDeclaringClass().getName(), getName()), ' ');
104            }
105
106            @Override
107            public String getQualifiedName() {
108                return f.getDeclaringClass().getName()+"."+getName();
109            }
110
111            @Override
112            public boolean isRoutable() {
113                return Modifier.isPublic(f.getModifiers());
114            }
115        };
116    }
117}
