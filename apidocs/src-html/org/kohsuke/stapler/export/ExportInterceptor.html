001package org.kohsuke.stapler.export;
002
003import java.io.IOException;
004import java.lang.reflect.InvocationTargetException;
005import java.util.logging.Level;
006import java.util.logging.Logger;
007
008/**
009 * Allows caller to intercept exporting of properties.
010 *
011 * Implementation can choose to ignore properties in case of failure during serialization.
012 *
013 * @author Vivek Pandey
014 * @author James Dumay
015 */
016public abstract class ExportInterceptor {
017
018    public static final Logger LOGGER = Logger.getLogger(ExportInterceptor.class.getName());
019
020    /**
021     * Constant to tell if return of {@link ExportInterceptor#getValue(Property, Object, ExportConfig)} should be skipped.
022     *
023     * Constant to skip serializaing a property in case of error
024     */
025    public static final Object SKIP = new Object();
026
027    /**
028     * Subclasses must call {@link Property#getValue(Object)}  to retrieve the property.
029     *
030     * If the subclass decides the value can be included in the request return the value
031     * otherwise, return {@link #SKIP}  to skip the property.
032     *
033     * @param property to get the value from model object
034     * @param model object with this property
035     * @return the value of the property, if {@link #SKIP} is returned, this property will be skipped
036     * @throws IOException if there was a problem with serialization that should prevent
037     *         the serialization from proceeding
038     * @see Exported#skipNull()
039     */
040    public abstract Object getValue(Property property, Object model, ExportConfig config) throws IOException;
041
042    public static final ExportInterceptor DEFAULT = new ExportInterceptor() {
043        @Override
044        public Object getValue(Property property, Object model, ExportConfig config) throws IOException {
045            try {
046                return property.getValue(model);
047            } catch (IllegalAccessException | InvocationTargetException e) {
048                if(config.isSkipIfFail()) {
049                    LOGGER.log(Level.WARNING,"Failed to get \"" + property.name + "\" from a " + model.getClass().getName(), e);
050                    return SKIP;
051                }
052                throw new IOException("Failed to write " + property.name + ":" + e.getMessage(), e);
053            }
054        }
055    };
056}
