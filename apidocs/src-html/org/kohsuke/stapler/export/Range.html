001package org.kohsuke.stapler.export;
002
003import com.google.common.collect.Iterators;
004
005import java.util.Arrays;
006import java.util.Collection;
007import java.util.Iterator;
008import java.util.List;
009
010/**
011 * Specifies the range in a collection.
012 *
013 * @author Kohsuke Kawaguchi
014 */
015public class Range {
016    public final int min;
017    public final int max;
018
019    public Range(int min, int max) {
020        this.min = min;
021        this.max = max;
022    }
023
024    public <T> List<T> apply(T[] a) {
025        return apply(Arrays.asList(a));
026    }
027
028    public <T> List<T> apply(List<T> s) {
029        if (max<s.size())   s = s.subList(0,max);
030        if (min>0)          s = s.subList(min,s.size());
031        return s;
032    }
033
034    public <T> Iterable<T> apply(final Collection<T> s) {
035        return apply((Iterable<T>)s);
036    }
037
038    public <T> Iterable<T> apply(final Iterable<T> s) {
039        if (s instanceof List) {
040            return apply((List<T>) s);
041        } else {
042            return new Iterable<T>() {
043                public Iterator<T> iterator() {
044                    Iterator<T> itr = s.iterator();
045                    itr = Iterators.limit(itr,max);
046                    if (min > 0) {
047                        for (int i = 0; i < min && itr.hasNext(); i++) {
048                            itr.next();
049                        }
050                    }
051                    return itr;
052                }
053            };
054        }
055    }
056
057    /**
058     * Range that includes natural numbers.
059     */
060    public static final Range ALL = new Range(0,Integer.MAX_VALUE);
061}
