024package org.kohsuke.stapler.export;
025
026import javax.annotation.Nonnull;
027import javax.annotation.Nullable;
028import java.io.IOException;
029import java.lang.reflect.Type;
030
031/**
032 * Receives the event callback on the model data to be exposed.
033 *
034 * <p>
035 * The call sequence is:
036 *
037 * <pre>
038 * EVENTS := type? startObject PROPERTY* endObject
039 * PROPERTY := name VALUE
040 * VALUE := valuePrimitive
041 *        | value
042 *        | valueNull
043 *        | startArray VALUE* endArray
044 *        | EVENTS
045 * </pre>
046 * 
047 * @author Kohsuke Kawaguchi
048 */
049public interface DataWriter {
050    void name(String name) throws IOException;
051
052    void valuePrimitive(Object v) throws IOException;
053    void value(String v) throws IOException;
054    void valueNull() throws IOException;
055
056    void startArray() throws IOException;
057    void endArray() throws IOException;
058
059    /**
060     * Augments the next {@link #startObject()} call by specifying the type information of that object.
061     *
062     * @param expected
063     *      The declared type of the variable that references this object.
064     *      Null if the object is not referenced by anyone, for example when it's the root.
065     * @param actual
066     *      The actual type of the object being written.
067     *      Null if the object is synthetic and has no valid Java type
068     */
069    default void type(@Nullable Type expected, @Nullable Class actual) throws IOException {}
070    void startObject() throws IOException;
071    void endObject() throws IOException;
072
073    default @Nonnull ExportConfig getExportConfig() {
074        return new ExportConfig();
075    }
076
077    /**
078     * Recommended property name to write out the 'type' parameter of {@link #type(Type,Class)}
079     */
080    String CLASS_PROPERTY_NAME = "_class";
081}
