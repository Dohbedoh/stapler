046/**
047 * Exposes one {@link Exported exposed property} of {@link ExportedBean} to
048 * {@link DataWriter}.
049 *  
050 * @author Kohsuke Kawaguchi
051 */
052public abstract class Property implements Comparable<Property> {
053
054    private static final Logger LOGGER = Logger.getLogger(Property.class.getName());
055
056    /**
057     * Name of the property.
058     */
059    public final String name;
060    final ModelBuilder owner;
061    /**
062     * Visibility depth level of this property.
063     *
064     * @see Exported#visibility()
065     */
066    public final int visibility;
067
068    /**
069     * Model to which this property belongs to.
070     * Never null.
071     */
072    public final Model parent;
073
074    /**
075     * @see Exported#inline()
076     */
077    public final boolean inline;
078
079    /**
080     * @see Exported#merge()
081     */
082    public final boolean merge;
083
084    /**
085     * @see Exported#skipNull()
086     */
087    private final boolean skipNull;
088
089    private String[] verboseMap;
090
091    private final Type type;
092
093    Property(Model parent, String name, Type type, Exported exported) {
094        this.parent = parent;
095        this.owner = parent.parent;
096        this.name = exported.name().length()>1 ? exported.name() : name;
097        this.type = type;
098        int v = exported.visibility();
099        if(v==0)
100            v = parent.defaultVisibility;
101        this.visibility = v;
102        this.inline = exported.inline();
103        this.merge = exported.merge();
104        this.skipNull = exported.skipNull();
105        String[] s = exported.verboseMap().split("/");
106        if (s.length<2)
107            this.verboseMap = null;
108        else
109            this.verboseMap = s;
110    }
111
112    public int compareTo(Property that) {
113        return this.name.compareTo(that.name);
114    }
115
116    public abstract Type getGenericType();
117    public abstract Class getType();
118
119    /**
120     * Gets the associated javadoc, if any, or null.
121     */
122    public abstract String getJavadoc();
123
124    /**
125     * Writes one property of the given object to {@link DataWriter}.
126     *
127     * @param pruner
128     *      Determines how to prune the object graph tree.
129     */
130    @SuppressWarnings("unchecked")
131    public void writeTo(Object object, TreePruner pruner, DataWriter writer) throws IOException {
132        TreePruner child = pruner.accept(object, this);
133        if (child==null)        return;
134
135        Object d = writer.getExportConfig().getExportInterceptor().getValue(this,object, writer.getExportConfig());
136
137        if ((d==null && skipNull) || d == ExportInterceptor.SKIP) { // don't write anything
138            return;
139        }
140        if (merge) {
141            // merged property will get all its properties written here
142            if (d != null) {
143                Class<?> objectType = d.getClass();
144                Model model = owner.getOrNull(objectType, parent.type, name);
145                if (model == null && !writer.getExportConfig().isSkipIfFail()) {
146                    throw new NotExportableException(objectType);
147                } else if (model != null) {
148                    model.writeNestedObjectTo(d, new FilteringTreePruner(parent.HAS_PROPERTY_NAME_IN_ANCESTRY,child), writer);
149                }
150            }
151        } else {
152            writer.name(name);
153            writeValue(type, d, child, writer);
154        }
155    }
156
157    /**
158     * @deprecated as of 1.139
159     */
160    public void writeTo(Object object, int depth, DataWriter writer) throws IOException {
161        writeTo(object,new ByDepth(depth),writer);
162    }
163
164    /**
165     * Writes one value of the property to {@link DataWriter}.
166     */
167    private void writeValue(Type expected, Object value, TreePruner pruner, DataWriter writer) throws IOException {
168        writeValue(expected,value,pruner,writer,writer.getExportConfig().isSkipIfFail());
169    }
170
171    private void writeBuffered(Type expected, Object value, TreePruner pruner, DataWriter writer) throws IOException {
172        BufferedDataWriter buffer = new BufferedDataWriter(writer.getExportConfig());
173        try {
174            writeValue(expected, value, pruner, buffer, true);
175            buffer.finished();
176        } catch (IOException x) {
177            if (x.getCause() instanceof InvocationTargetException) {
178                LOGGER.log(Level.WARNING, "skipping export of " + value, x);
179            }
180        }
181        buffer.commit(writer);
182    }
183
184    /**
185     * Writes one value of the property to {@link DataWriter}.
186     */
187    @SuppressWarnings("unchecked")
188    private void writeValue(Type expected, Object value, TreePruner pruner, DataWriter writer, boolean skipIfFail) throws IOException {
189        if(value==null) {
190            writer.valueNull();
191            return;
192        }
193
194        if(value instanceof CustomExportedBean) {
195            writeValue(expected,((CustomExportedBean)value).toExportedObject(),pruner,writer);
196            return;
197        }
198
199        Class c = value.getClass();
200
201        Model model = owner.getOrNull(c, parent.type, name);
202        if (model == null) {
203            if(STRING_TYPES.contains(c)) {
204                writer.value(value.toString());
205                return;
206            }
207            if(PRIMITIVE_TYPES.contains(c)) {
208                writer.valuePrimitive(value);
209                return;
210            }
211            Class act = c.getComponentType();
212            if (act !=null) { // array
213                Range r = pruner.getRange();
214                writer.startArray();
215                if (value instanceof Object[]) {
216                    // typical case
217                    for (Object item : r.apply((Object[]) value)) {
218                        writeBuffered(act, item, pruner, writer);
219                    }
220                } else {
221                    // more generic case
222                    int len = Math.min(r.max, Array.getLength(value));
223                    for (int i=r.min; i<len; i++) {
224                        writeBuffered(act, Array.get(value, i), pruner, writer);
225                    }
226                }
227                writer.endArray();
228                return;
229            }
230            if(value instanceof Iterable) {
231                writer.startArray();
232                Type expectedItemType = Types.getTypeArgument(expected, 0, null);
233                for (Object item : pruner.getRange().apply((Iterable) value)) {
234                    writeBuffered(expectedItemType, item, pruner, writer);
235                }
236                writer.endArray();
237                return;
238            }
239            if(value instanceof Map) {
240                if (verboseMap!=null) {// verbose form
241                    writer.startArray();
242                    for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
243                        BufferedDataWriter buffer = new BufferedDataWriter(writer.getExportConfig());
244                        try {
245                            writeStartObjectNullType(buffer);
246                            buffer.name(verboseMap[0]);
247                            writeValue(null, e.getKey(), pruner, buffer);
248                            buffer.name(verboseMap[1]);
249                            writeValue(null, e.getValue(), pruner, buffer);
250                            buffer.endObject();
251                            buffer.finished();
252                        } catch (IOException x) {
253                            if (x.getCause() instanceof InvocationTargetException) {
254                                LOGGER.log(Level.WARNING, "skipping export of " + e, x);
255                            }
256                        }
257                        buffer.commit(writer);
258                    }
259                    writer.endArray();
260                } else {// compact form
261                    writeStartObjectNullType(writer);
262                    for (Map.Entry e : ((Map<?,?>) value).entrySet()) {
263                        BufferedDataWriter buffer = new BufferedDataWriter(writer.getExportConfig());
264                        try {
265                            buffer.name(e.getKey().toString());
266                            writeValue(null, e.getValue(), pruner, buffer);
267                            buffer.finished();
268                        } catch (IOException x) {
269                            if (x.getCause() instanceof InvocationTargetException) {
270                                LOGGER.log(Level.WARNING, "skipping export of " + e, x);
271                            }
272                        }
273                        buffer.commit(writer);
274                    }
275                    writer.endObject();
276                }
277                return;
278            }
279            if(value instanceof Date) {
280                writer.valuePrimitive(((Date) value).getTime());
281                return;
282            }
283            if(value instanceof Calendar) {
284                writer.valuePrimitive(((Calendar) value).getTimeInMillis());
285                return;
286            }
287            if(value instanceof Enum) {
288                writer.value(value.toString());
289                return;
290            }
291
292            if (skipIfFail) {
293                writer.startObject();
294                writer.endObject();
295                return;
296            }
297
298            throw new NotExportableException(c);
299        } else {
300            writer.type(expected, value.getClass());
301            writer.startObject();
302            model.writeNestedObjectTo(value, pruner, writer);
303            writer.endObject();
304        }
305    }
306
307    private static class BufferedDataWriter implements DataWriter {
308        private final ExportConfig exportConfig;
309
310        private BufferedDataWriter(ExportConfig exportConfig) {
311            this.exportConfig = exportConfig;
312        }
313
314        @Override
315        public @Nonnull ExportConfig getExportConfig() {
316            return exportConfig;
317        }
318
319        enum Op {name, valuePrimitive, value, valueNull, startArray, endArray, type, startObject, endObject}
320        static class Step {
321            final Op op;
322            final Object[] args;
323            Step(Op op, Object... args) {
324                this.op = op;
325                this.args = args;
326            }
327        }
328        final List<Step> steps = new ArrayList<Step>();
329        boolean finished = false;
330        @Override
331        public void name(String name) throws IOException {
332            steps.add(new Step(Op.name, name));
333        }
334        @Override
335        public void valuePrimitive(Object v) throws IOException {
336            steps.add(new Step(Op.valuePrimitive, v));
337        }
338        @Override
339        public void value(String v) throws IOException {
340            steps.add(new Step(Op.value, v));
341        }
342        @Override
343        public void valueNull() throws IOException {
344            steps.add(new Step(Op.valueNull));
345        }
346        @Override
347        public void startArray() throws IOException {
348            steps.add(new Step(Op.startArray));
349        }
350        @Override
351        public void endArray() throws IOException {
352            steps.add(new Step(Op.endArray));
353        }
354        @Override
355        public void type(Type expected, Class actual) throws IOException {
356            steps.add(new Step(Op.type, expected, actual));
357        }
358        @Override
359        public void startObject() throws IOException {
360            steps.add(new Step(Op.startObject));
361        }
362        @Override
363        public void endObject() throws IOException {
364            steps.add(new Step(Op.endObject));
365        }
366        private void finished() {
367            finished = true;
368        }
369        void commit(DataWriter w) throws IOException {
370            if (!finished) {
371                return;
372            }
373            for (Step step : steps) {
374                switch (step.op) {
375                case name:
376                    w.name((String) step.args[0]);
377                    break;
378                case valuePrimitive:
379                    w.valuePrimitive(step.args[0]);
380                    break;
381                case value:
382                    w.value((String) step.args[0]);
383                    break;
384                case valueNull:
385                    w.valueNull();
386                    break;
387                case startArray:
388                    w.startArray();
389                    break;
390                case endArray:
391                    w.endArray();
392                    break;
393                case type:
394                    w.type((Type) step.args[0], (Class) step.args[1]);
395                    break;
396                case startObject:
397                    w.startObject();
398                    break;
399                case endObject:
400                    w.endObject();
401                    break;
402                default:
403                    throw new AssertionError();
404                }
405            }
406        }
407    }
408
409    private void writeStartObjectNullType(DataWriter writer) throws IOException {
410        writer.type(null,null);
411        writer.startObject();
412    }
