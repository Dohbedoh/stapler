134    /**
135     * Calls {@link #bindAndInvoke(Object, StaplerRequest, StaplerResponse, Object...)} and then
136     * optionally serve the response object.
137     *
138     * @return
139     *      true if the request was dispatched and processed. false if the dispatch was cancelled
140     *      and the search for the next request handler should continue. An exception is thrown
141     *      if the request was dispatched but the processing failed.
142     */
143    boolean bindAndInvokeAndServeResponse(Object node, RequestImpl req, ResponseImpl rsp, Object... headArgs) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
144        try {
145            Object r = bindAndInvoke(node, req, rsp, headArgs);
146            if (getReturnType() != void.class)
147                renderResponse(req, rsp, node, r);
148            return true;
149        } catch (CancelRequestHandlingException _) {
150            return false;
151        } catch (InvocationTargetException e) {
152            // exception as an HttpResponse
153            Throwable te = e.getTargetException();
154            if (te instanceof CancelRequestHandlingException)
155                return false;
156            if (renderResponse(req,rsp,node,te))
157                return true;    // exception rendered the response
158            throw e;    // unprocessed exception
159        }
160    }
161
162    static boolean renderResponse(RequestImpl req, ResponseImpl rsp, Object node, Object ret) throws IOException, ServletException {
163        for (HttpResponseRenderer r : req.stapler.getWebApp().getResponseRenderers())
164            if (r.generateResponse(req,rsp,node,ret))
165                return true;
166        return false;
167    }

169    /**
170     * Use the given arguments as the first N arguments,
171     * then figure out the rest of the arguments by looking at parameter annotations,
172     * then finally call {@link #invoke}.
173     */
174    Object bindAndInvoke(Object o, StaplerRequest req, StaplerResponse rsp, Object... headArgs) throws IllegalAccessException, InvocationTargetException, ServletException {
175        Class[] types = getParameterTypes();
176        Annotation[][] annotations = getParameterAnnotations();
177        String[] parameterNames = getParameterNames();
178
179        Object[] arguments = new Object[types.length];
180
181        // fill in the first N arguments
182        System.arraycopy(headArgs,0,arguments,0,headArgs.length);
183
184        try {
185            // find the rest of the arguments. either known types, or with annotations
186            for( int i=headArgs.length; i<types.length; i++ ) {
187                Class t = types[i];
188                if(t==StaplerRequest.class || t==HttpServletRequest.class) {
189                    arguments[i] = req;
190                    continue;
191                }
192                if(t==StaplerResponse.class || t==HttpServletResponse.class) {
193                    arguments[i] = rsp;
194                    continue;
195                }
196
197                // if the databinding method is provided, call that
198                Function binder = PARSE_METHODS.getUnchecked(t);
199                if (binder!=RETURN_NULL) {
200                    arguments[i] = binder.bindAndInvoke(null,req,rsp);
201                    continue;
202                }
203                
204                arguments[i] = AnnotationHandler.handle(req,annotations[i],
205                    i<parameterNames.length ? parameterNames[i] : null,
206                    t);
207            }
208        } catch (IllegalArgumentException e) {
209            throw new IllegalArgumentException("Failed to invoke "+getDisplayName(),e);
210        }
211
212        return invoke(req, rsp, o,arguments);
213    }

215    /**
216     * Computing map that discovers the static 'fromStapler' method from a class.
217     * The discovered method will be returned as a Function so that the invocation can do parameter injections.
218     */
219    private static final LoadingCache<Class,Function> PARSE_METHODS;
220    private static final Function RETURN_NULL;
221
222    static {
223        try {
224            RETURN_NULL = new StaticFunction(Function.class.getMethod("returnNull"));
225        } catch (NoSuchMethodException e) {
226            throw new AssertionError(e);    // impossible
227        }
228
229        PARSE_METHODS = CacheBuilder.newBuilder().weakKeys().build(new CacheLoader<Class,Function>() {
230            public Function load(Class from) {
231                // MethdFunction for invoking a static method as a static method
232                FunctionList methods = new ClassDescriptor(from).methods.name("fromStapler");
233                switch (methods.size()) {
234                case 0: return RETURN_NULL;
235                default:
236                    throw new IllegalArgumentException("Too many 'fromStapler' methods on "+from);
237                case 1:
238                    Method m = ((MethodFunction)methods.get(0)).m;
239                    return new MethodFunction(m) {
240                        @Override
241                        public Class[] getParameterTypes() {
242                            return m.getParameterTypes();
243                        }
244
245                        @Override
246                        public Type[] getGenericParameterTypes() {
247                            return m.getGenericParameterTypes();
248                        }
249
250                        @Override
251                        public Annotation[][] getParameterAnnotations() {
252                            return m.getParameterAnnotations();
253                        }
254
255                        @Override
256                        public Object invoke(StaplerRequest req, StaplerResponse rsp, Object o, Object... args) throws IllegalAccessException, InvocationTargetException {
257                            return m.invoke(null,args);
258                        }
259                    };
260                }
261            }
262        });
263    }

275    final Function wrapByInterceptors(AnnotatedElement m) {
276        try {
277            Function f = this;
278            for (Annotation a : m.getAnnotations()) {
279                final InterceptorAnnotation ia = a.annotationType().getAnnotation(InterceptorAnnotation.class);
280                if (ia!=null) {
281                    try {
282                        Interceptor i = ia.value().newInstance();
283                        switch (ia.stage()) {
284                        case SELECTION:
285                            f = new SelectionInterceptedFunction(f,i);
286                            break;
287                        case PREINVOKE:
288                            f = new PreInvokeInterceptedFunction(f,i);
289                            break;
290                        }
291                    } catch (InstantiationException e) {
292                        throw (Error)new InstantiationError("Failed to instantiate interceptor for "+f.getDisplayName()).initCause(e);
293                    } catch (IllegalAccessException e) {
294                        throw (Error)new IllegalAccessError("Failed to instantiate interceptor for "+f.getDisplayName()).initCause(e);
295                    }
296                }
297            }
298
299            return f;
300        } catch (LinkageError e) {
301            // running in JDK 1.4
302            return this;
303        }
304    }

310    private abstract static class MethodFunction extends Function {
311        protected final Method m;
312        private volatile MethodHandle handle;
313
314        private volatile String[] names;
315
316        public MethodFunction(Method m) {
317            this.m = m;
318            // defer the resolution of MethodHandle so that a Function can be built to represent a non-public method
319        }
320
321        public final String getName() {
322            return m.getName();
323        }
324
325        public final String getDisplayName() {
326            return m.toGenericString();
327        }
328
329        @Override
330        public String getSignature() {
331            String prefix = isStatic() ? "staticMethod" : "method";
332            String value = StringUtils.join(Arrays.asList(prefix, m.getDeclaringClass().getName(), getName()), ' ');
333            if (getParameterTypes().length > 0) {
334                value += " " + StringUtils.join(Arrays.stream(getParameterTypes()).map(Class::getName).collect(Collectors.toList()), ' ');
335            }
336            return value;
337        }
338
339        @Override
340        public boolean isStatic() {
341            return Modifier.isStatic(m.getModifiers());
342        }
343
344        @Override
345        public String getQualifiedName() {
346            return m.getDeclaringClass().getName()+'.'+getName();
347        }
348
349        public <A extends Annotation> A getAnnotation(Class<A> annotation) {
350            return m.getAnnotation(annotation);
351        }
352
353        public Annotation[] getAnnotations() {
354            return m.getAnnotations();
355        }
356
357        public final String[] getParameterNames() {
358            if(names==null)
359                names = ClassDescriptor.loadParameterNames(m);
360            return names;
361        }
362
363        @Override
364        public Class getReturnType() {
365            return m.getReturnType();
366        }
367
368        @Override
369        public Class[] getCheckedExceptionTypes() {
370            return m.getExceptionTypes();
371        }
372    
373        @Override 
374        public Class getDeclaringClass() {
375            return m.getDeclaringClass();
376        }
377    
378        protected MethodHandle handle() {
379            if (handle==null) {
380                handle = MethodHandleFactory.get(m);
381            }
382            return handle;
383        }
384
385        @Override
386        public Object invoke(StaplerRequest req, StaplerResponse rsp, Object o, Object... args) throws IllegalAccessException, InvocationTargetException {
387            Object[] arguments;
388            if (Modifier.isStatic(m.getModifiers())) {
389                arguments = args;
390            } else {
391                arguments = new Object[args.length + 1];
392                arguments[0] = o;
393                System.arraycopy(args, 0, arguments, 1, args.length);
394            }
395            try {
396                return handle().invokeWithArguments(arguments);
397            } catch (WrongMethodTypeException x) {
398                LOGGER.log(Level.WARNING, handle + " failed on " + o + "." + m + Arrays.toString(arguments), x);
399            } catch (Throwable throwable) {
400                throw new InvocationTargetException(throwable);
401            }
402            return m.invoke(o, args);
403        }
404    }

427    /**
428     * Instance method that overrides other instance methods where we join annotations.
429     */
430    static final class OverridingInstanceFunction extends InstanceFunction {
431        // the last one takes precedence
432        private final List<Method> methods;
433
434        public OverridingInstanceFunction(List<Method> m) {
435            super(m.get(0));
436            methods = m;
437        }
438
439        @Override
440        public <A extends Annotation> A getAnnotation(Class<A> annotation) {
441            for (Method m : methods) {
442                A a = m.getAnnotation(annotation);
443                if (a!=null)    return a;
444            }
445            return null;
446        }
447
448        @Override
449        public Annotation[] getAnnotations() {
450            Annotation[] x = new Annotation[0];
451            for (Method m : methods) {
452                x = union(x, m.getAnnotations());
453            }
454            return x;
455        }
456
457        public Annotation[][] getParameterAnnotations() {
458            Annotation[][] all = null;
459            for (Method m : methods) {
460                Annotation[][] next = m.getParameterAnnotations();
461                if (all==null) {
462                    all = next;
463                } else {
464                    for (int i = 0; i < next.length; i++) {
465                        all[i] = union(all[i], next[i]);
466                    }
467                }
468            }
469            return all;
470        }
471    }

473    /**
474     * Static methods on the wrapper type.
475     */
476    static final class StaticFunction extends MethodFunction {
477        public StaticFunction(Method m) {
478            super(m);
479        }
480
481        public Class[] getParameterTypes() {
482            Class[] p = m.getParameterTypes();
483            Class[] r = new Class[p.length-1];
484            System.arraycopy(p,1,r,0,r.length);
485            return r;
486        }
487
488        @Override
489        public Type[] getGenericParameterTypes() {
490            Type[] p = m.getGenericParameterTypes();
491            Type[] r = new Type[p.length-1];
492            System.arraycopy(p,1,r,0,r.length);
493            return r;
494        }
495
496        public Annotation[][] getParameterAnnotations() {
497            Annotation[][] a = m.getParameterAnnotations();
498            Annotation[][] r = new Annotation[a.length-1][];
499            System.arraycopy(a,1,r,0,r.length);
500            return r;
501        }
502    }
