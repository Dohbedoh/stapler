044/**
045 * Created one instance each for a {@link Klass},
046 * that retains some useful cache about a class and its views.
047 *
048 * @author Kohsuke Kawaguchi
049 * @see WebApp#getMetaClass(Klass)
050 */
051public class MetaClass extends TearOffSupport {
052    /**
053     * This meta class wraps this class
054     *
055     * @deprecated as of 1.177
056     *      Use {@link #klass}. If you really want the Java class representation, use {@code klass.toJavaClass()}.
057     */
058    public final Class clazz;
059
060    public final Klass<?> klass;
061
062    /**
063     * {@link MetaClassLoader} that wraps {@code clazz.getClassLoader()}.
064     * Null if the class is loaded by the bootstrap classloader.
065     */
066    public final MetaClassLoader classLoader;
067
068    public final List<Dispatcher> dispatchers = new ArrayList<>();
069
070    /**
071     * Base metaclass.
072     * Note that <tt>baseClass.clazz==clazz.getSuperClass()</tt>
073     */
074    public final MetaClass baseClass;
075
076    /**
077     * {@link WebApp} that owns this meta class.
078     */
079    public final WebApp webApp;
080
081    /**
082     * If there's a method annotated with @PostConstruct, that {@link MethodRef} object, linked
083     * to the list of the base class.
084     */
085    private volatile SingleLinkedList<MethodRef> postConstructMethods;
086
087    /*package*/ MetaClass(WebApp webApp, Klass<?> klass) {
088        this.clazz = klass.toJavaClass();
089        this.klass = klass;
090        this.webApp = webApp;
091        this.baseClass = webApp.getMetaClass(klass.getSuperClass());
092        this.classLoader = MetaClassLoader.get(clazz.getClassLoader());
093        buildDispatchers();
094    }
095
096    /**
097     * Build {@link #dispatchers}.
098     *
099     * <p>
100     * This is the meat of URL dispatching. It looks at the class
101     * via reflection and figures out what URLs are handled by who.
102     */
103    /*package*/ void buildDispatchers() {
104        this.dispatchers.clear();
105        KlassDescriptor<?> node = new KlassDescriptor(klass);
106
107        dispatchers.add(new DirectoryishDispatcher());
108
109        if (HttpDeletable.class.isAssignableFrom(clazz))
110            dispatchers.add(new HttpDeletableDispatcher());
111
112        // check action <obj>.do<token>(...) and other WebMethods
113        for (Function f : node.methods.webMethods()) {
114            WebMethod a = f.getAnnotation(WebMethod.class);
115
116            String[] names;
117            if(a!=null && a.name().length>0)   names=a.name();
118            else    names=new String[]{camelize(f.getName().substring(2))}; // 'doFoo' -> 'foo'
119
120            for (String name : names) {
121                final Function ff = f.contextualize(new WebMethodContext(name));
122                if (name.length()==0) {
123                    dispatchers.add(new IndexDispatcher(ff));
124                } else {
125                    dispatchers.add(new NameBasedDispatcher(name) {
126                        public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
127                            Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s", ff.getName());
128                            if (traceable())
129                                trace(req, rsp, "-> <%s>.%s(...)", node, ff.getName());
130                            return ff.bindAndInvokeAndServeResponse(node, req, rsp);
131                        }
132
133                        public String toString() {
134                            return ff.getQualifiedName() + "(...) for url=/" + name + "/...";
135                        }
136                    });
137                }
138            }
139        }
140
141        // check action <obj>.doIndex(...)
142        for (Function f : node.methods.name("doIndex")) {
143            dispatchers.add(new IndexDispatcher(f.contextualize(new WebMethodContext(""))));
144        }
145
146        // JavaScript proxy method invocations for <obj>js<token>
147        // reacts only to a specific content type
148        for (Function f : node.methods.prefix("js") ) {
149            String name = camelize(f.getName().substring(2)); // jsXyz -> xyz
150            f = f.contextualize(new JavaScriptMethodContext(name));
151            dispatchers.add(new JavaScriptProxyMethodDispatcher(name, f));
152        }
153
154        // JavaScript proxy method with @JavaScriptMethod
155        // reacts only to a specific content type
156        for( final Function f : node.methods.annotated(JavaScriptMethod.class) ) {
157            JavaScriptMethod a = f.getAnnotation(JavaScriptMethod.class);
158
159            String[] names;
160            if(a!=null && a.name().length>0)   names=a.name();
161            else    names=new String[]{f.getName()};
162
163            for (String name : names)
164                dispatchers.add(new JavaScriptProxyMethodDispatcher(name,f.contextualize(new JavaScriptMethodContext(name))));
165        }
166
167        for (Facet f : webApp.facets)
168            f.buildViewDispatchers(this, dispatchers);
169
170        for (Facet f : webApp.facets)
171            f.buildIndexDispatchers(this, dispatchers);
172
173        Dispatcher d = IndexHtmlDispatcher.make(webApp.context, clazz);
174        if (d!=null)
175            dispatchers.add(d);
176
177        // check public properties of the form NODE.TOKEN
178        for (final FieldRef f : node.fields) {
179            dispatchers.add(new NameBasedDispatcher(f.getName()) {
180                final String role = getProtectedRole(f);
181                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException {
182                    if(role!=null && !req.isUserInRole(role))
183                        throw new IllegalAccessException("Needs to be in role "+role);
184
185                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s", f.getName());
186                    if(traceable())
187                        traceEval(req,rsp,node,f.getName());
188                    req.getStapler().invoke(req, rsp, f.get(node));
189                    return true;
190                }
191                public String toString() {
192                    return String.format("%1$s for url=/%2$s/...",f.getQualifiedName(),f.getName());
193                }
194            });
195        }
196
197        FunctionList getMethods = node.methods.prefix("get");
198
199        // check public selector methods of the form NODE.getTOKEN()
200        for (Function f : getMethods.signature()) {
201            if(f.getName().length()<=3)
202                continue;
203
204            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
205            final Function ff = f.contextualize(new TraversalMethodContext(name));
206
207            dispatchers.add(new NameBasedDispatcher(name) {
208                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
209                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s()", ff.getName());
210                    if(traceable())
211                        traceEval(req,rsp,node,ff.getName()+"()");
212                    req.getStapler().invoke(req,rsp, ff.invoke(req, rsp, node));
213                    return true;
214                }
215                public String toString() {
216                    return String.format("%1$s() for url=/%2$s/...",ff.getQualifiedName(),name);
217                }
218            });
219        }
220
221        // check public selector methods of the form static NODE.getTOKEN(StaplerRequest)
222        for (Function f : getMethods.signature(StaplerRequest.class)) {
223            if(f.getName().length()<=3)
224                continue;
225            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
226            final Function ff = f.contextualize(new TraversalMethodContext(name));
227            dispatchers.add(new NameBasedDispatcher(name) {
228                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
229                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s(...)", ff.getName());
230                    if(traceable())
231                        traceEval(req,rsp,node,ff.getName()+"(...)");
232                    req.getStapler().invoke(req,rsp, ff.invoke(req, rsp, node, req));
233                    return true;
234                }
235                public String toString() {
236                    return String.format("%1$s(StaplerRequest) for url=/%2$s/...",ff.getQualifiedName(),name);
237                }
238            });
239        }
240
241        // check public selector methods <obj>.get<Token>(String)
242        for (Function f : getMethods.signature(String.class)) {
243            if(f.getName().length()<=3)
244                continue;
245            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
246            final Function ff = f.contextualize(new TraversalMethodContext(name));
247            dispatchers.add(new NameBasedDispatcher(name,1) {
248                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
249                    String token = req.tokens.next();
250                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s(String)", ff.getName());
251                    if(traceable())
252                        traceEval(req,rsp,node,ff.getName()+"(\""+token+"\")");
253                    req.getStapler().invoke(req,rsp, ff.invoke(req, rsp, node,token));
254                    return true;
255                }
256                public String toString() {
257                    return String.format("%1$s(String) for url=/%2$s/TOKEN/...",ff.getQualifiedName(),name);
258                }
259            });
260        }
261
262        // check public selector methods <obj>.get<Token>(int)
263        for (Function f : getMethods.signature(int.class)) {
264            if(f.getName().length()<=3)
265                continue;
266            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
267            final Function ff = f.contextualize(new TraversalMethodContext(name));
268            dispatchers.add(new NameBasedDispatcher(name,1) {
269                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
270                    int idx = req.tokens.nextAsInt();
271                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s(int)", ff.getName());
272                    if(traceable())
273                        traceEval(req,rsp,node,ff.getName()+"("+idx+")");
274                    req.getStapler().invoke(req,rsp, ff.invoke(req, rsp, node,idx));
275                    return true;
276                }
277                public String toString() {
278                    return String.format("%1$s(int) for url=/%2$s/N/...",ff.getQualifiedName(),name);
279                }
280            });
281        }
282
283        // check public selector methods <obj>.get<Token>(long)
284        // TF: I'm sure these for loop blocks could be dried out in some way.
285        for (Function f : getMethods.signature(long.class)) {
286            if(f.getName().length()<=3)
287                continue;
288            String name = camelize(f.getName().substring(3)); // 'getFoo' -> 'foo'
289            final Function ff = f.contextualize(new TraversalMethodContext(name));
290            dispatchers.add(new NameBasedDispatcher(name,1) {
291                public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException, IllegalAccessException, InvocationTargetException {
292                    long idx = req.tokens.nextAsLong();
293                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s(long)", ff.getName());
294                    if(traceable())
295                        traceEval(req,rsp,node,ff.getName()+"("+idx+")");
296                    req.getStapler().invoke(req,rsp, ff.invoke(req, rsp, node,idx));
297                    return true;
298                }
299                public String toString() {
300                    return String.format("%1$s(long) for url=/%2$s/N/...",ff.getQualifiedName(),name);
301                }
302            });
303        }
304
305        if (klass.isArray()) {
306            dispatchers.add(new Dispatcher() {
307                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
308                    if(!req.tokens.hasMore())
309                        return false;
310                    try {
311                        int index = req.tokens.nextAsInt();
312                        Dispatcher.anonymizedTraceEval(req, rsp, node, "%s[idx]");
313                        if (traceable())
314                            traceEval(req, rsp, node, "", "[" + index + "]");
315                        req.getStapler().invoke(req, rsp, klass.getArrayElement(node, index));
316                        return true;
317                    } catch (IndexOutOfBoundsException e) {
318                        if(traceable())
319                            trace(req,rsp,"-> IndexOutOfRange");
320                        rsp.sendError(SC_NOT_FOUND);
321                        return true;
322                    } catch (NumberFormatException e) {
323                        return false; // try next
324                    }
325                }
326                public String toString() {
327                    return "Array look-up for url=/N/...";
328                }
329            });
330        }
331
332        if(klass.isMap()) {
333            dispatchers.add(new Dispatcher() {
334                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IOException, ServletException {
335                    if(!req.tokens.hasMore())
336                        return false;
337                    try {
338                        String key = req.tokens.peek();
339                        Dispatcher.anonymizedTraceEval(req, rsp, node, "%s: Map access");
340                        if(traceable())
341                            traceEval(req,rsp,"",".get(\""+key+"\")");
342
343                        Object item = klass.getMapElement(node,key);
344                        if(item!=null) {
345                            req.tokens.next();
346                            req.getStapler().invoke(req,rsp,item);
347                            return true;
348                        } else {
349                            // otherwise just fall through
350                            if(traceable())
351                                trace(req,rsp,"Map.get(\""+key+"\")==null. Back tracking.");
352                            return false;
353                        }
354                    } catch (NumberFormatException e) {
355                        return false; // try next
356                    }
357                }
358                public String toString() {
359                    return "Map.get(String) look-up for url=/TOKEN/...";
360                }
361            });
362        }
363
364        // TODO: check if we can route to static resources
365        // which directory shall we look up a resource from?
366
367        for (Facet f : webApp.facets)
368            f.buildFallbackDispatchers(this, dispatchers);
369
370        // check public selector methods <obj>.getDynamic(<token>,...)
371        for (Function f : getMethods.signatureStartsWith(String.class).name("getDynamic")) {
372            final Function ff = f.contextualize(new TraversalMethodContext(TraversalMethodContext.DYNAMIC));
373            dispatchers.add(new Dispatcher() {
374                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, IOException, ServletException {
375                    if(!req.tokens.hasMore())
376                        return false;
377                    String token = req.tokens.next();
378                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#getDynamic(...)");
379                    if(traceable())
380                        traceEval(req,rsp,node,"getDynamic(\""+token+"\",...)");
381
382                    Object target = ff.bindAndInvoke(node, req,rsp, token);
383                    if(target!=null) {
384                        req.getStapler().invoke(req,rsp, target);
385                        return true;
386                    } else {
387                        if(traceable())
388                            // indent:    "-> evaluate(
389                            trace(req,rsp,"            %s.getDynamic(\"%s\",...)==null. Back tracking.",node,token);
390                        req.tokens.prev(); // cancel the next effect
391                        return false;
392                    }
393                }
394                public String toString() {
395                    return String.format("%s(String,StaplerRequest,StaplerResponse) for url=/TOKEN/...",ff.getQualifiedName());
396                }
397            });
398        }
399
400        // check action <obj>.doDynamic(...)
401        for (Function f : node.methods.name("doDynamic")) {
402            final Function ff = f.contextualize(new WebMethodContext(WebMethodContext.DYNAMIC));
403            dispatchers.add(new Dispatcher() {
404                public boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
405                    Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#doDynamic(...)");
406                    if(traceable())
407                        trace(req,rsp,"-> <%s>.doDynamic(...)",node);
408                    return ff.bindAndInvokeAndServeResponse(node,req,rsp);
409                }
410                public String toString() {
411                    return String.format("%s(StaplerRequest,StaplerResponse) for any URL",ff.getQualifiedName());
412                }
413            });
414        }
415    }

417    /**
418     * Returns all the methods in the ancestry chain annotated with {@link PostConstruct}
419     * from those defined in the derived type toward those defined in the base type.
420     *
421     * Normally invocation requires visiting the list in the reverse order.
422     * @since 1.220
423     */
424    public SingleLinkedList<MethodRef> getPostConstructMethods() {
425        if (postConstructMethods ==null) {
426            SingleLinkedList<MethodRef> l = baseClass==null ? SingleLinkedList.<MethodRef>empty() : baseClass.getPostConstructMethods();
427
428            for (MethodRef mr : klass.getDeclaredMethods()) {
429                if (mr.hasAnnotation(PostConstruct.class)) {
430                    l = l.grow(mr);
431                }
432            }
433            postConstructMethods = l;
434        }
435        return postConstructMethods;
436    }

438    private String getProtectedRole(FieldRef f) {
439        try {
440            LimitedTo a = f.getAnnotation(LimitedTo.class);
441            return (a!=null)?a.value():null;
442        } catch (LinkageError e) {
443            return null;    // running in JDK 1.4
444        }
445    }

447    @Override
448    public String toString() {
449        return "MetaClass["+klass+"]";

452    private static String camelize(String name) {
453        return Character.toLowerCase(name.charAt(0))+name.substring(1);
454    }

456    private static class JavaScriptProxyMethodDispatcher extends NameBasedDispatcher {
457        private final Function f;
458
459        public JavaScriptProxyMethodDispatcher(String name, Function f) {
460            super(name, 0);
461            this.f = f;
462        }
463
464        public boolean doDispatch(RequestImpl req, ResponseImpl rsp, Object node) throws IllegalAccessException, InvocationTargetException, ServletException, IOException {
465            if (!req.isJavaScriptProxyCall())
466                return false;
467
468            req.stapler.getWebApp().getCrumbIssuer().validateCrumb(req,req.getHeader("Crumb"));
469
470            Dispatcher.anonymizedTraceEval(req, rsp, node, "%s#%s", f.getName());
471            if(traceable())
472                trace(req,rsp,"-> <%s>.%s(...)",node, f.getName());
473
474            JSONArray jsargs = JSONArray.fromObject(IOUtils.toString(req.getReader()));
475            Object[] args = new Object[jsargs.size()];
476            Class[] types = f.getParameterTypes();
477            Type[] genericTypes = f.getGenericParameterTypes();
478            if (args.length != types.length) {
479                throw new IllegalArgumentException("argument count mismatch between " + jsargs + " and " + Arrays.toString(genericTypes));
480            }
481
482            for (int i=0; i<args.length; i++)
483                args[i] = req.bindJSON(genericTypes[i],types[i],jsargs.get(i));
484
485            return f.bindAndInvokeAndServeResponse(node,req,rsp,args);
486        }
487
488        public String toString() {
489            return f.getQualifiedName()+"(...) for url=/"+name+"/...";
490        }
491    }
