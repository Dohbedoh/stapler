001package org.kohsuke.stapler.interceptor;
002
003import java.io.IOException;
004import java.io.PrintWriter;
005
006import org.kohsuke.stapler.ForwardToView;
007import org.kohsuke.stapler.StaplerRequest;
008import org.kohsuke.stapler.StaplerResponse;
009
010import java.lang.annotation.Retention;
011import java.lang.annotation.Target;
012import java.lang.reflect.InvocationTargetException;
013import java.util.ServiceLoader;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import javax.annotation.CheckForNull;
019import javax.servlet.ServletException;
020import javax.servlet.http.HttpServletResponse;
021import org.kohsuke.stapler.HttpResponses;
022import org.kohsuke.stapler.verb.POST;
023
024/**
025 * Requires the request to be a POST.
026 *
027 * <p>
028 * When the current request has a non-matching HTTP method (such as 'GET'), this annotation
029 * will send a failure response instead of searching for other matching web methods.
030 *
031 * @author Kohsuke Kawaguchi
032 * @since 1.180
033 * @see POST
034 */
035@Retention(RUNTIME)
036@Target({METHOD,FIELD})
037@InterceptorAnnotation(RequirePOST.Processor.class)
038public @interface RequirePOST {
039
040    /**
041     * Allows customizing the error page shown when an annotated method is called with the wrong HTTP method.
042     */
043    interface ErrorCustomizer {
044        /**
045         * Return the {@link ForwardToView} showing a custom error page for {@link RequirePOST} annotated methods. This is
046         * typically used to show a form with a "Try again using POST" button.
047         *
048         * <p>Implementations are looked up using {@link java.util.ServiceLoader}, the first implementation to return a non-null value will be used.</p>
049         */
050        @CheckForNull ForwardToView getForwardView();
051    }
052
053    public static class Processor extends Interceptor {
054        @Override
055        public Object invoke(StaplerRequest request, StaplerResponse response, Object instance, Object[] arguments)
056                throws IllegalAccessException, InvocationTargetException, ServletException {
057            if (!request.getMethod().equals("POST")) {
058                for (ErrorCustomizer handler : ServiceLoader.load(ErrorCustomizer.class, request.getWebApp().getClassLoader())) {
059                    ForwardToView forwardToView = handler.getForwardView();
060                    if (forwardToView != null) {
061                        throw new InvocationTargetException(forwardToView.with("requestURL", request.getRequestURLWithQueryString().toString()));
062                    }
063                }
064                throw new InvocationTargetException(new HttpResponses.HttpResponseException() {
065                    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
066                        rsp.setStatus(HttpServletResponse.SC_METHOD_NOT_ALLOWED);
067                        rsp.addHeader("Allow", "POST");
068                        rsp.setContentType("text/html");
069                        PrintWriter w = rsp.getWriter();
070                        w.println("<html><head><title>POST required</title></head><body>");
071                        w.println("POST is required for " + target.getQualifiedName() + "<br>");
072                        w.println("<form method='POST'><input type='submit' value='Try POSTing'></form>");
073                        w.println("</body></html>");
074                    }
075                });
076            }
077            return target.invoke(request, response, instance, arguments);
078        }
079    }
080}
