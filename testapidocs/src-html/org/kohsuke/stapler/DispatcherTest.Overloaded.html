052            return HttpResponses.text("Got GET");

061    /**
062     * Tests the dispatching of WebMethod based on verb
063     */
064    public void testVerbMatch() throws Exception {
065        WebClient wc = new WebClient();
066
067        check(wc, HttpMethod.GET);
068        check(wc, HttpMethod.POST);
069        try {
070            check(wc, HttpMethod.DELETE);
071            fail("There's no route for DELETE");
072        } catch (FailingHttpStatusCodeException e) {
073            assertEquals(404, e.getStatusCode());
074        }
075    }
076
077    private void check(WebClient wc, HttpMethod m) throws java.io.IOException {
078        TextPage p = wc.getPage(new WebRequestSettings(new URL(url, "verbMatch/"), m));
079        assertEquals("Got " + m.name(), p.getContent());
080    }

199    public final PutInheritance putInheritance = new PutInheritanceImpl();
200    public abstract class PutInheritance {
201        @WebMethod(name="foo") @PUT
202        public abstract HttpResponse doBar(StaplerRequest req) throws IOException;
203
204        @POST
205        public HttpResponse doAcme(StaplerRequest req) throws IOException {
206            return HttpResponses.text("POST: " + IOUtils.toString(req.getInputStream()));
207        }
208    }
209
210    public class PutInheritanceImpl extends PutInheritance{
211        @Override
212        public HttpResponse doBar(StaplerRequest req) throws IOException {
213            return HttpResponses.text(IOUtils.toString(req.getInputStream()) + " World!");
214        }
215    }
216
217    public void testPutInheritance() throws Exception {
218        WebClient wc = new WebClient();
219
220        // the request should get to the overriding method and it should still see all the annotations in the base type
221        WebRequestSettings wrs = new WebRequestSettings(new URL(url, "putInheritance/foo"), HttpMethod.PUT);
222        wrs.setRequestBody("Hello");
223        TextPage p = wc.getPage(wrs);
224        assertEquals("Hello World!", p.getContent());
225
226        // doBar is a web method for 'foo', so bar endpoint shouldn't respond
227        try {
228            wc.getPage(new URL(url, "putInheritance/bar"));
229            fail();
230        } catch (FailingHttpStatusCodeException e) {
231            assertEquals(404,e.getStatusCode());
232        }
233
234        //Invoke Post as well
235        wrs = new WebRequestSettings(new URL(url, "putInheritance/acme"), HttpMethod.POST);
236        wrs.setRequestBody("Hello");
237        p = wc.getPage(wrs);
238        assertEquals("POST: Hello", p.getContent());
239    }

278    public abstract class RequirePostOnBase {
279        int hit;
280        @RequirePOST
281        public abstract void doSomething();
282    }

285        @Override
286        public void doSomething() {
287            hit++;
288        }
289    }
290
291    public void testRequirePostOnBase() throws Exception {
292        WebClient wc = new WebClient();
293        URL url = new URL(this.url, "requirePostOnBase/something");
294
295        try {
296            wc.getPage(url);
297            fail();
298        } catch (FailingHttpStatusCodeException e) {
299            assertEquals(HttpServletResponse.SC_METHOD_NOT_ALLOWED, e.getStatusCode());
300        }
301
302        // POST should succeed
303        wc.getPage(new WebRequestSettings(url, HttpMethod.POST));
304        assertEquals(1, requirePostOnBase.hit);
305    }
