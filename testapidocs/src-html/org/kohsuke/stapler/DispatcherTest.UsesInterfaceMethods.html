391    public final StaplerProxyImpl staplerProxyOK = new StaplerProxyImpl(new IndexPage());
392    public final StaplerProxyImpl staplerProxyFail = new StaplerProxyImpl(null);
393
394    public class IndexPage {
395        public void doIndex(StaplerResponse rsp) {
396            throw HttpResponses.ok();
397        }
398    }
399
400    public class StaplerProxyImpl implements StaplerProxy {
401        private Object target;
402        public int counter = 0;
403        public StaplerProxyImpl(Object target) {
404            this.target = target;
405        }
406
407        @Override
408        public Object getTarget() {
409            counter++;
410            return target;
411        }
412
413        public void doIndex(StaplerResponse rsp) {
414            if (target != this) {
415                throw new IllegalStateException("should not be called");
416            }
417        }
418    }
419
420
421    public void testStaplerProxy() throws Exception {
422        WebClient wc = new WebClient();
423        Page p = wc.getPage(new URL(url, "staplerProxyOK"));
424        assertEquals(200, p.getWebResponse().getStatusCode());
425
426        try {
427            p = wc.getPage(new URL(url, "staplerProxyFail"));
428            fail("expected failure");
429        } catch (FailingHttpStatusCodeException ex) {
430            assertEquals(404, ex.getStatusCode());
431        }
432        assertTrue(staplerProxyOK.counter > 0);
433        assertTrue(staplerProxyFail.counter > 0);
434    }
