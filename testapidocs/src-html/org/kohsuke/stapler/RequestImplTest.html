024package org.kohsuke.stapler;
025
026import org.apache.commons.fileupload.FileItem;
027import org.apache.http.entity.ContentType;
028import org.apache.http.entity.mime.MultipartEntityBuilder;
029import org.junit.Assert;
030import org.junit.Test;
031
032import javax.servlet.ReadListener;
033import javax.servlet.ServletException;
034import javax.servlet.ServletInputStream;
035import java.io.ByteArrayInputStream;
036import java.io.ByteArrayOutputStream;
037import java.io.File;
038import java.io.IOException;
039import java.util.Arrays;
040import java.util.Collections;
041import java.util.Enumeration;
042
043/**
044 * @author <a href="mailto:tom.fennelly@gmail.com">tom.fennelly@gmail.com</a>
045 */
046public class RequestImplTest {
047
048    @Test
049    public void test_multipart_formdata() throws IOException, ServletException {
050        final Stapler stapler = new Stapler();
051        final byte[] buf = generateMultipartData();
052        final ByteArrayInputStream is = new ByteArrayInputStream(buf);
053        final MockRequest mockRequest = new MockRequest() {
054            @Override
055            public String getContentType() {
056                return "multipart/form-data; boundary=mpboundary";
057            }
058            @Override
059            public String getCharacterEncoding() {
060                return "UTF-8";
061            }
062            @Override
063            public int getContentLength() {
064                return buf.length;
065            }
066
067            @Override
068            public Enumeration getParameterNames() {
069                return Collections.enumeration(Arrays.asList("p1"));
070            }
071
072            @Override
073            public ServletInputStream getInputStream() throws IOException {
074                return new ServletInputStream() {
075                    @Override
076                    public int read() throws IOException {
077                        return is.read();
078                    }
079                    @Override
080                    public int read(byte[] b) throws IOException {
081                        return is.read(b);
082                    }
083                    @Override
084                    public int read(byte[] b, int off, int len) throws IOException {
085                        return is.read(b, off, len);
086                    }
087                    @Override
088                    public boolean isFinished() {
089                        return is.available() != 0;
090                    }
091                    @Override
092                    public boolean isReady() {
093                        return true;
094                    }
095                    @Override
096                    public void setReadListener(ReadListener readListener) {
097                        // ignored
098                    }
099                    
100                };
101            }
102        };
103        
104        RequestImpl request = new RequestImpl(stapler, mockRequest, Collections.<AncestorImpl>emptyList(), null);
105
106        // Check that we can get the Form Fields. See https://github.com/stapler/stapler/issues/52
107        Assert.assertEquals("text1_val", request.getParameter("text1"));
108        Assert.assertEquals("text2_val", request.getParameter("text2"));
109        
110        // Check that we can get the file
111        FileItem fileItem = request.getFileItem("pomFile");
112        Assert.assertNotNull(fileItem);
113        
114        // Check getParameterValues
115        Assert.assertEquals("text1_val", request.getParameterValues("text1")[0]);
116        
117        // Check getParameterNames
118        Assert.assertTrue(Collections.list(request.getParameterNames()).contains("p1"));
119        Assert.assertTrue(Collections.list(request.getParameterNames()).contains("text1"));
120        
121        // Check getParameterMap
122        Assert.assertTrue(request.getParameterMap().containsKey("text1"));        
123    }
124
125    private byte[] generateMultipartData() throws IOException {
126        MultipartEntityBuilder reqEntityBuilder = MultipartEntityBuilder.create();
127
128        reqEntityBuilder.setBoundary("mpboundary");
129        reqEntityBuilder.addBinaryBody("pomFile", new File("./pom.xml"), ContentType.TEXT_XML, "pom.xml");
130        reqEntityBuilder.addTextBody("text1", "text1_val");
131        reqEntityBuilder.addTextBody("text2", "text2_val");
132
133        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
134        try {
135            reqEntityBuilder.build().writeTo(outputStream);
136            outputStream.flush();
137            return outputStream.toByteArray();
138        } finally {
139            outputStream.close();
140        }
141    }
142}
