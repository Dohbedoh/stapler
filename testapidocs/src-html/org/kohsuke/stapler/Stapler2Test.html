027package org.kohsuke.stapler;
028
029import com.gargoylesoftware.htmlunit.WebClient;
030import com.gargoylesoftware.htmlunit.WebResponse;
031import java.net.URL;
032import javax.servlet.http.HttpServletResponse;
033import org.jvnet.hudson.test.For;
034import org.jvnet.hudson.test.Issue;
035import org.kohsuke.stapler.test.JettyTestCase;
036
037@For(Stapler.class) // but StaplerTest is not a JettyTestCase
038public class Stapler2Test extends JettyTestCase {
039
040    @Issue("SECURITY-390")
041    public void testTraceXSS() throws Exception {
042        WebClient wc = new WebClient();
043        wc.setThrowExceptionOnFailingStatusCode(false);
044        WebResponse rsp;
045        Dispatcher.TRACE = true;
046        try {
047            rsp = wc.getPage(new URL(this.url, "thing/<button>/x")).getWebResponse();
048        } finally {
049            Dispatcher.TRACE = false;
050        }
051        assertEquals(HttpServletResponse.SC_NOT_FOUND, rsp.getStatusCode());
052        String html = rsp.getContentAsString();
053        assertTrue(html, html.contains("&lt;button&gt;"));
054        assertFalse(html, html.contains("<button>"));
055    }
056    public Object getThing(String name) {
057        return name;
058    }
059
060}
