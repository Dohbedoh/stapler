001package org.kohsuke.stapler;
002
003import junit.framework.TestCase;
004import org.junit.Assert;
005import org.kohsuke.stapler.lang.Klass;
006
007public class MetaClassTest extends TestCase {
008    public void testGetObjectProhibited() throws Exception {
009        MetaClass metaClass = new MetaClass(new WebApp(new MockServletContext()), Klass.java(Object.class));
010
011        // ensure no getClass dispatcher for Object
012        Assert.assertFalse(metaClass.dispatchers.stream().filter(d -> d instanceof NameBasedDispatcher).anyMatch(d -> ((NameBasedDispatcher) d).name.equals("class")));
013
014        // in fact, there should be no name based dispatchers at all
015        Assert.assertFalse(metaClass.dispatchers.stream().anyMatch(d -> d instanceof NameBasedDispatcher));
016    }
017}
